// Helper to create a mock successful task update chain
function mockSuccessfulTaskUpdate(currentStatusId = 'status-old') {
  const selectChain = {
    eq: vi.fn().mockReturnValue({
      single: vi.fn().mockResolvedValue({
        data: {
          status_id: currentStatusId,
          updated_at: new Date(Date.now() - 10000).toISOString(), // Server updated 10s ago
        },
        error: null,
      }),
    }),
  };

  const updateChain = {
    eq: vi.fn().mockResolvedValue({ error: null }),
  };

  mockSupabaseClient.from.mockImplementation((table: string) => {
    if (table === 'tasks') {
      return {
        select: vi.fn().mockReturnValue(selectChain),
        update: vi.fn().mockReturnValue(updateChain),
      };
    }
    return {};
  });
}

  const selectChain = {

      const onProgress = (progress: SyncProgress) => {
        progressCalls.push({ ...progress });

      await processAllMutations(onProgress);

      expect(progressCalls.length).toBeGreaterThanOrEqual(2);

        task_id: 'task-3',
        status_id: 's3',
        previous_status_id: 'status-old',

  describe('processAllMutations - conflict detection', () => {
    it('should detect conflict when server status differs from previous_status_id', async () => {
      // Server has different status than what we expected (previous_status_id)
      // AND server was updated after our mutation was created
      const serverUpdatedAt = new Date(Date.now() + 1000); // 1s in future
      mockConflictScenario('status-different', serverUpdatedAt);

      const mutation = await queueStatusMutation({
        task_id: 'task-123',
        status_id: 'status-new',
        previous_status_id: 'status-old', // We expected this
      });

      const result = await processAllMutations();

      expect(result.conflicts).toHaveLength(1);
      expect(result.conflicts[0].mutationId).toBe(mutation!.id);
      expect(result.conflicts[0].conflict).toBeDefined();
      expect(result.conflicts[0].conflict!.local_value).toBe('status-new');
      expect(result.conflicts[0].conflict!.server_value).toBe('status-different');

      // Mutation should still exist, marked as conflict
      const fetched = await getMutation(mutation!.id);
      expect(fetched).toBeDefined();
      expect(fetched!.status).toBe('conflict');
    });

    it('should NOT detect conflict when server status matches previous_status_id', async () => {
      // Server has the status we expected - no conflict
      mockSuccessfulTaskUpdate('status-old');

      const mutation = await queueStatusMutation({
        task_id: 'task-123',
        status_id: 'status-new',
        previous_status_id: 'status-old', // Matches server
      });

      const result = await processAllMutations();

      expect(result.conflicts).toHaveLength(0);
      expect(result.status).toBe('synced');

      // Mutation should be deleted (successful sync)
      const fetched = await getMutation(mutation!.id);
      expect(fetched).toBeUndefined();
    });

    it('should NOT detect conflict when server is older than mutation', async () => {
      // Server has different status BUT was updated before our mutation
      const serverUpdatedAt = new Date(Date.now() - 60000); // 1 minute ago

      const selectChain = {
        eq: vi.fn().mockReturnValue({
          single: vi.fn().mockResolvedValue({
            data: {
              status_id: 'status-different',
              updated_at: serverUpdatedAt.toISOString(),
            },
            error: null,
          }),
        }),
      };

      const updateChain = {
        eq: vi.fn().mockResolvedValue({ error: null }),
      };

      mockSupabaseClient.from.mockImplementation((table: string) => {
        if (table === 'tasks') {
          return {
            select: vi.fn().mockReturnValue(selectChain),
            update: vi.fn().mockReturnValue(updateChain),
          };
        }
        return {};
      });

      // Wait a bit to ensure mutation created_at is after serverUpdatedAt
      await new Promise((r) => setTimeout(r, 10));

      const mutation = await queueStatusMutation({
        task_id: 'task-123',
        status_id: 'status-new',
        previous_status_id: 'status-old',
      });

      const result = await processAllMutations();

      // No conflict because our mutation is newer than server update
      expect(result.conflicts).toHaveLength(0);
    });

    it('should skip conflict check when force_overwrite is true', async () => {
      // Server has different status but force_overwrite bypasses conflict check
      const serverUpdatedAt = new Date(Date.now() + 1000);

      const selectChain = {
        eq: vi.fn().mockReturnValue({
          single: vi.fn().mockResolvedValue({
            data: {
              status_id: 'status-different',
              updated_at: serverUpdatedAt.toISOString(),
            },
            error: null,
          }),
        }),
      };

      const updateChain = {
        eq: vi.fn().mockResolvedValue({ error: null }),
      };

      mockSupabaseClient.from.mockImplementation((table: string) => {
        if (table === 'tasks') {
          return {
            select: vi.fn().mockReturnValue(selectChain),
            update: vi.fn().mockReturnValue(updateChain),
          };
        }
        return {};
      });

      // Create mutation and simulate conflict resolution with 'local' choice
      const mutation = await queueStatusMutation({
        task_id: 'task-123',
        status_id: 'status-new',
        previous_status_id: 'status-old',
      });

      // Simulate marking as conflict then resolving with 'local'
      const { markMutationConflict, resolveConflict } = await import('../mutation-queue');
      await markMutationConflict(mutation!.id, {
        detected_at: new Date().toISOString(),
        local_value: 'status-new',
        server_value: 'status-different',
        server_updated_at: serverUpdatedAt.toISOString(),
        field_name: 'status',
      });
      await resolveConflict(mutation!.id, 'local'); // Sets force_overwrite: true

      const result = await processAllMutations();

      // Should succeed without conflict despite server difference
      expect(result.conflicts).toHaveLength(0);
      expect(result.errors).toHaveLength(0);
    });
  });

                    error: null,

              }),

              eq: vi.fn().mockResolvedValue({ error: null }),

        return {};
      });

      await queueStatusMutation({
        task_id: 'task-1',
        status_id: 's1',
        previous_status_id: 'status-old',
      });
      await new Promise((r) => setTimeout(r, 10));
      await queueStatusMutation({
        task_id: 'task-2',
        status_id: 's2',
        previous_status_id: 'status-old',
      });

      const result = await processAllMutations();

      // Both were processed, one failed, one succeeded

      expect(result.current).toBe(2);
      expect(result.errors).toHaveLength(1);

        task_id: 'task-1',
        status_id: 's1',
        previous_status_id: 'status-old',
      });
      await queueStatusMutation({
        task_id: 'task-2',
        status_id: 's2',
        previous_status_id: 'status-old',
      });

      const result = await processAllMutations();

      expect(result.errors).toHaveLength(2);
      expect(result.status).toBe('error');
    });
  });
