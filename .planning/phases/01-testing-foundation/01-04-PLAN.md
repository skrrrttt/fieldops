```typescript
/**
 * Delete all mutations that were successfully synced
 * Call this periodically or after sync to clean up completed mutations
 */
export async function clearSyncedMutations(): Promise<number> {
  if (!isIndexedDBAvailable()) return 0;

  const db = getDB();
  // Delete all mutations with 'synced' status
  const count = await db.pending_mutations.where('status').equals('synced').delete();
  return count;
}
```

Note: The return type changes from `Promise<void>` to `Promise<number>` to report how many were deleted.
