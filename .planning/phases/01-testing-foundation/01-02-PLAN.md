  describe('FIFO ordering', () => {
    it('should return pending mutations in FIFO order (oldest first)', async () => {
      // Create mutations with explicit timestamps to ensure order
      const m1 = await queueStatusMutation({
        task_id: 'task-1',
        status_id: 's1',
        previous_status_id: 's0',
      });

      // Small delay to ensure different timestamps
      await new Promise((r) => setTimeout(r, 10));

      const m2 = await queueStatusMutation({
        task_id: 'task-2',
        status_id: 's2',
        previous_status_id: 's1',
      });

      await new Promise((r) => setTimeout(r, 10));

      const m3 = await queueStatusMutation({
        task_id: 'task-3',
        status_id: 's3',
        previous_status_id: 's2',
      });

      const pending = await getPendingMutations();

      expect(pending).toHaveLength(3);
      // FIFO: oldest first (m1 created first)
      expect(pending[0].id).toBe(m1!.id);
      expect(pending[1].id).toBe(m2!.id);
      expect(pending[2].id).toBe(m3!.id);
    });

    it('should only return pending status mutations from getPendingMutations', async () => {
      const m1 = await queueStatusMutation({
        task_id: 'task-1',
        status_id: 's1',
        previous_status_id: 's0',
      });
      const m2 = await queueStatusMutation({
        task_id: 'task-2',
        status_id: 's2',

      expect(updated!.error_message).toBe('Connection refused');
